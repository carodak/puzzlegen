 43        """
 44        Check if a position is within the grid bounds.
 45
 46        Args:
 47
 48            row (int): Row index.
 49
 50            col (int): Column index.
 51
 52        Returns:
 53            bool: True if position is valid, False otherwise.

 61        Check if a block is present at the given position.

 63        Args:

 67            col (int): Column index.

100    def check_valid_placement(self, row, col, color):
101        """
102        Check if placing a block at the given position is valid according to the rules (no matches are allowed during initial setup).
103
104        Args:
105
106            row (int): Row index.
107
108            col (int): Column index.
109
110            color (str): Color of the block.
111
112        Returns:
113            bool: True if placement is valid, False otherwise.
114        """
115        if self.is_similar_block(row + 1, col, color) and self.is_similar_block(row + 2, col, color):
116            return False
117        if self.is_similar_block(row, col-1, color) and self.is_similar_block(row, col+1, color):
118            return False
119        if self.is_similar_block(row, col-1, color) and self.is_similar_block(row, col-2, color):
120            return False
121        if self.is_similar_block(row, col+1, color) and self.is_similar_block(row, col+2, color):
122            return False
123        return True

125    def get_highest_positions(self):
126        """
127        Get the highest (topmost) block positions for each column.
128
129        Returns:
130            list: List of (row, col) tuples representing the highest block in each column.
131        """
132        highest_positions = {}
133        for position in self.set_blocks:
134            row, col = position
135            if col not in highest_positions or row < highest_positions[col][0]:
136                highest_positions[col] = (row, col)

140    def select_random_highest_position(self):
141        """
142        Select a random position above the highest block in a random column or return an empty position with a gap.
143
144        Returns:
145            tuple: (row, col) position above a random highest block, or an empty position with gap if none found.
146        """
147        highest_block_positions = self.get_highest_positions()
148        highest_block_positions_filtered = [(row, col) for row, col in highest_block_positions if row != 0]
149        if not highest_block_positions_filtered:
150            return self.select_empty_positions_with_gap()
151        random_position = random.choice(highest_block_positions_filtered)
152        return (random_position[0]-1,random_position[1])

154    def get_empty_positions_with_gap(self):
155        """
156        Get empty positions in the bottomost row that respect the minimum gap between blocks.
157
158        Returns:
159            list: List of (row, col) tuples for valid empty positions.
160        """
161        last_row = self.grid_size[0] - 1
162        empty_positions_with_gap = []
163        highest_positions = self.get_highest_positions()
164
165        for row, col in highest_positions:
166            adjacent_cols = [col + self.blocks_gap, col + self.blocks_gap + 1, col - self.blocks_gap, col - self.blocks_gap - 1]
167            adjacent_cols = [adj_col for adj_col in adjacent_cols if adj_col >= 0 and adj_col < self.grid_size[1]]
168
169            for adj_col in adjacent_cols:
170                adjacent_position = (last_row, adj_col)
171                if adjacent_position not in self.set_blocks:
172                    empty_positions_with_gap.append(adjacent_position)
173
174        return empty_positions_with_gap

177        """
178        Select a random empty position in the last row that respects the minimum gap.

180        Returns:
181            tuple: (row, col) position.
182        """
183        empty_positions_with_gap = self.get_empty_positions_with_gap()
184        if not empty_positions_with_gap:

189    def initialize_grid(self):
190        """
191        Initialize the grid with blocks according to the rules and constraints.
192        
193        The method performs the following steps:
194          - Determines the number of blocks to place per color using `assign_blocks_per_color`.
195          - Creates a list of blocks to place, respecting the color distribution.
196          - Iteratively places each block in the grid, either stacking on existing blocks or
197            placing in empty positions that respect minimum spacing (gap) constraints.
198          - The first block is placed near the center top of the grid.
199          - For subsequent blocks, the placement is either stacked on top of existing blocks (with a
200            probability defined by `stack_probability`) or placed in valid empty positions.
201          - Each potential placement is validated by `check_valid_placement` to avoid rule violations (e.g., no immediate matches).
202          - Once placed, blocks are recorded in `self.set_blocks` with their positions as keys.
203          - Finally, `self.set_blocks` is sorted by row to maintain a consistent order.
204
205        This process ensures the initial puzzle grid respects the game's placement rules
206        and sets up a valid starting configuration for the puzzle.
207
208        Returns:
209            None
210        """
211        blocks_per_color = assign_blocks_per_color(self.nb_blocks, self.colors)
212        blocks_to_place = [color for color, count in blocks_per_color for i in range(count)]
213        while blocks_to_place:
214            placed = False
215            while not(placed):
216                stacked = False
217                if random.random() < self.stack_probability:
218                    stacked = True
219                random_block_index = random.randrange(len(blocks_to_place))
220                random_block_color = blocks_to_place[random_block_index]
221                if not self.set_blocks:
222                    random_column_grid = math.floor(self.grid_size[1]/2)
223                    selected_row = self.grid_size[0]-1
224                elif not stacked:
225                    random_position = self.select_empty_positions_with_gap()
226                    random_column_grid = random_position[1]
227                    selected_row = random_position[0]
228                if stacked and self.set_blocks:
229                    random_position = self.select_random_highest_position()
230                    random_column_grid = random_position[1]
231                    selected_row = random_position[0]
232
233                if self.check_valid_placement(selected_row, random_column_grid, random_block_color):
234                    block = Block(random_block_color)
235                    self.set_blocks[(selected_row, random_column_grid)] = block
236                    placed = True
237            blocks_to_place.pop(random_block_index)
238        self.set_blocks = sort_blocks_by_rows(self.set_blocks)

244        Args:

247        self.set_blocks = set_blocks

250    def init_graphical_grid(set_blocks, grid_size):
251        """
252        Create a graphical representation of the puzzle grid using matplotlib.
253
254        This method builds a 2D color grid from the set of placed blocks, then draws it using 
255        matplotlib by rendering each cell as a colored rectangle. The Y-axis is flipped so that 
256        row 0 appears at the bottom, consistent with a bottom-up grid orientation.
257
258        Args:
259
260            set_blocks (dict): Dictionary mapping (row, col) positions to Block objects. Each block
261                                contains a `color` attribute used for rendering.
262
263            grid_size (tuple): A (rows, columns) tuple indicating the dimensions of the grid.
264
265        Returns:
266        
267            tuple: (fig, ax, grid)
268                - fig (matplotlib.figure.Figure): The matplotlib figure object.
269                - ax (matplotlib.axes.Axes): The matplotlib axis with rendered grid.
270                - grid (list of lists): 2D list of cell colors representing the grid state.
271        """
272        grid = [['white' for _ in range(grid_size[1])] for _ in range(grid_size[0])]
273        for position, block in set_blocks.items():
274            row, col = position
275            grid[row][col] = block.color
276
277        fig, ax = plt.subplots()
278        ax.axis('off')
279        for row in range(grid_size[0]):
280            for col in range(grid_size[1]):
281                color = grid[row][col]
282                rect = plt.Rectangle((col, grid_size[0] - row - 1), 1, 1, facecolor=color, edgecolor='black')
283                ax.add_patch(rect)
284
285        return fig, ax, grid

306            str(self.set_blocks)

192        Initialize the grid with blocks according to the rules and constraints.
193        
194        The method performs the following steps:
195          - Determines the number of blocks to place per color using `assign_blocks_per_color`.
196          - Creates a list of blocks to place, respecting the color distribution.
197          - Iteratively places each block in the grid, either stacking on existing blocks or
198            placing in empty positions that respect minimum spacing (gap) constraints.
199          - The first block is placed near the center top of the grid.
200          - For subsequent blocks, the placement is either stacked on top of existing blocks (with a
201            probability defined by `stack_probability`) or placed in valid empty positions.
202          - Each potential placement is validated by `check_valid_placement` to avoid rule violations (e.g., no immediate matches).
203          - Once placed, blocks are recorded in `self.set_blocks` with their positions as keys.
204          - Finally, `self.set_blocks` is sorted by row to maintain a consistent order.
205
206        This process ensures the initial puzzle grid respects the game's placement rules
207        and sets up a valid starting configuration for the puzzle.
208
209        Returns:
210            None
211        """
212        blocks_per_color = assign_blocks_per_color(self.nb_blocks, self.colors)
213        blocks_to_place = [color for color, count in blocks_per_color for i in range(count)]
214        while blocks_to_place:
215            placed = False
216            while not(placed):
217                stacked = False
218                if random.random() < self.stack_probability:
219                    stacked = True
220                random_block_index = random.randrange(len(blocks_to_place))
221                random_block_color = blocks_to_place[random_block_index]
222                if not self.set_blocks:
223                    random_column_grid = math.floor(self.grid_size[1]/2)
224                    selected_row = self.grid_size[0]-1
225                elif not stacked:
226                    random_position = self.select_empty_positions_with_gap()
227                    random_column_grid = random_position[1]
228                    selected_row = random_position[0]
229                if stacked and self.set_blocks:
230                    random_position = self.select_random_highest_position()
231                    random_column_grid = random_position[1]
232                    selected_row = random_position[0]

237                    placed = True
238            blocks_to_place.pop(random_block_index)
239        self.set_blocks = sort_blocks_by_rows(self.set_blocks)
